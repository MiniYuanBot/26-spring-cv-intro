\makeatletter
% Global list spacing: eliminate vertical gaps in itemize/enumerate
\def\@listI{%
    \leftmargin\leftmargini
    \parsep 0pt
    \topsep 0pt
    \itemsep 0pt plus 0.1pt
}
\let\@listii\@listI
\let\@listiii\@listI

% 1. Body text: 7pt font, 7.5pt leading
\renewcommand\normalsize{%
    \@setfontsize\normalsize{7pt}{7.5pt}%
    \abovedisplayskip 2\p@ \@plus1\p@ \@minus1\p@
    \belowdisplayskip \abovedisplayskip
    \abovedisplayshortskip \z@ \@plus1\p@
    \belowdisplayshortskip 1\p@ \@plus1\p@ \@minus1\p@
    \let\@listi\@listI
    \zihao{6}} % CJK size 6 (~7.5pt), paired with Latin body size
\normalsize

% 2. Math size declarations: {text}{display}{script}{scriptscript}
%    First argument must match the body font size (7pt)
\DeclareMathSizes{7}{7}{5}{4}

% 3. Small text: 6.5pt font, 7pt leading
\renewcommand\small{%
    \@setfontsize\small{6.5pt}{7pt}%
    \abovedisplayskip 1.5\p@ \@plus1\p@ \@minus1\p@
    \belowdisplayskip \abovedisplayskip
    \abovedisplayshortskip \z@ \@plus1\p@
    \belowdisplayshortskip 0.5\p@ \@plus1\p@ \@minus1\p@
    \let\@listi\@listI
    \zihao{6}}

% 4. Section headings — use \def to preserve *-version handling via \@startsection
\def\section{%
  \@startsection{section}{1}{0mm}
  {.2ex plus .1ex minus .05ex}
  {.1ex plus .05ex}
  {\color{myblue}\sffamily\fontsize{7.5pt}{8pt}\selectfont\bfseries\zihao{6}}}
\def\subsection{%
    \@startsection{subsection}{2}{0mm}
    {.1ex plus .05ex minus .02ex}
    {.05ex plus .02ex}
    {\sffamily\bfseries\fontsize{7pt}{7.5pt}\selectfont\zihao{6}}}
\def\subsubsection{%
    \@startsection{subsubsection}{3}{0mm}
    {.05ex plus .02ex minus .01ex}
    {.01ex plus .01ex}
    {\sffamily\fontsize{6.5pt}{7pt}\selectfont\zihao{6}}}
\makeatother

% ========== Custom Commands ==========
% Compact image insert (no float, works inside multicols)
\newcommand{\imgbox}[2][\linewidth]{%
    \par
    \vspace{0.2mm}%
    {\centering
        \includegraphics[width=#1]{#2}%
        \par
    }
    \vspace{0.2mm}%
}

% 2-up: \imgpair[width]{left-img}{right-img}
\newcommand{\imgpair}[3][0.48\linewidth]{%
    \par\noindent
    \begin{minipage}[t]{#1}%
        \centering\includegraphics[width=\hsize]{#2}%
    \end{minipage}%
    \hfill
    \begin{minipage}[t]{#1}%
        \centering\includegraphics[width=\hsize]{#3}%
    \end{minipage}%
    \par\vspace{0.2mm}%
}

% 3-up: \imgtriple[width]{img1}{img2}{img3}
\newcommand{\imgtriple}[4][0.31\linewidth]{%
    \par\noindent
    \begin{minipage}[t]{#1}%
        \centering\includegraphics[width=\hsize]{#2}%
    \end{minipage}%
    \hfill
    \begin{minipage}[t]{#1}%
        \centering\includegraphics[width=\hsize]{#3}%
    \end{minipage}%
    \hfill
    \begin{minipage}[t]{#1}%
        \centering\includegraphics[width=\hsize]{#4}%
    \end{minipage}%
    \par\vspace{0.2mm}%
}
